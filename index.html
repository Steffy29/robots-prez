Tinygo code example

              
	gp1 := machine.GP1
	pwm0b := machine.PWM0
	pwm0b.Configure(machine.PWMConfig{
		Period: period,
	})
	ch1, err1 := pwm0b.Channel(gp1)
	if err1 != nil {
		println(err1.Error())
		return
	}
	position := 1
  
  
  
	for {
		for i := 0; i < 4; i++ {
			if position == 16 {
				position = 1
			}
			pwm0b.Set(ch1, pwm0b.Top()/uint32(position))
			time.Sleep(time.Millisecond * 10)
			position = position + 4
		}
	}
